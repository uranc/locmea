116  def cmp_fwd_matrix(self, electrode_pos, voxels, p_sigma=0.3):
117  vx, vy, vz = voxels
118  elx, ely, elz = electrode_pos.T
119  # Flatten
120  vx, vy, vz = vx.flatten(), vy.flatten(), vz.flatten()

124  n_elz = np.unique(elz).shape[0]
125  n_el = n_elx * n_ely * n_elz
126  n_v = vx.shape[0]
127  # Check position vector formats
128  if n_el != elx.shape[0] or n_el != ely.shape[0] \
129  or n_el != elz.shape[0] or n_el <= 0:
130  print 'Electrode coordinates are wrong'
131  fwd_matrix = np.zeros((n_el, n_v))
132  for el in np.arange(0, n_el):
133  for v in np.arange(0, n_v):
134  fwd_matrix[el, v] = np.sqrt((elx[el] - vx[v]) ** 2 +
135  (ely[el] - vy[v]) ** 2 +
136  (elz[el] - vz[v]) ** 2)
137  return 1. / (fwd_matrix * (4. * np.pi * p_sigma))
